   159        /*! Lists of cache modes for L1/L2 caches */
   160        enum Mode {
   161            Mode_FREEZE,    /*! No new cache lines are allocated */
   162            Mode_BYPASS,    /*! All access result in long-distance access */
   163            Mode_NORMAL     /*! Normal operation of cache */

   166        /*! Level 1 cache size type definition. Can be used for both L1D & L1P */
   167        enum L1Size {
   168            L1Size_0K = 0,  /*! Amount of cache is 0K, Amount of SRAM is 32K */
   169            L1Size_4K = 1,  /*! Amount of cache is 4K, Amount of SRAM is 28K */
   170            L1Size_8K = 2,  /*! Amount of cache is 8K, Amount of SRAM is 24K */
   171            L1Size_16K = 3, /*! Amount of cache is 16K, Amount of SRAM is 16K */
   172            L1Size_32K = 4  /*! Amount of cache is 32K, Amount of SRAM is 0K */
   173        };
   174    
   175        /*! Level 2 cache size type definition. */
   176        enum L2Size {
   177            L2Size_0K = 0,   /*! L2 is all SRAM */
   178            L2Size_32K = 1,  /*! Amount of cache is 32K */
   179            L2Size_64K = 2,  /*! Amount of cache is 64K */
   180            L2Size_128K = 3, /*! Amount of cache is 128K */
   181            L2Size_256K = 4, /*! Amount of cache is 256K */
   182            L2Size_512K = 5, /*! Amount of cache is 512K */
   183            L2Size_1024K = 6 /*! Amount of cache is 1024K */
   184        };
   185    
   186        /*! MAR register setting type definition. */
   187        enum Mar {
   188            Mar_DISABLE = 0, /*! The Permit Copy bit of MAR register is disabled */
   189            Mar_ENABLE = 1   /*! The Permit Copy bit of MAR register is enabled */
   190        };

   192        /*! Structure for specifying all cache sizes. */

   428         *  ======== inv ========
   429         *  Invalidate the range of memory within the specified starting address and
   430         *  byte count.  The range of addresses operated on gets quantized to whole
   431         *  cache lines in each cache.  All cache lines in range are invalidated in
   432         *  L1P cache.  All cache lines in range are invalidated in L1D cache.
   433         *  All cache lines in range are invaliated in L2 cache.
   434         */
   435        override Void inv(Ptr blockPtr, SizeT byteCnt, Bits16 type, Bool wait);
   436    
   437        /*!
   438         *  ======== wb ========
   439         *  Writes back the range of memory within the specified starting address
   440         *  and byte count.  The range of addresses operated on gets quantized to
   441         *  whole cache lines in each cache.  There is no effect on L1P cache.
   442         *  All cache lines within the range are left valid in L1D cache and the
   443         *  data within the range in L1D cache will be written back to L2 or
   444         *  external memory. All cache lines within the range are left valid in
   445         *  L2 cache and the data within the range in L2 cache will be written back
   446         *  to external memory.
   447         */
   448        override Void wb(Ptr blockPtr, SizeT byteCnt, Bits16 type, Bool wait);
   449    
   450        /*!
   451         *  ======== wbInv ========
   452         *  Writes back and invalidates the range of memory within the specified
   453         *  starting address and byte count.  The range of addresses operated on
   454         *  gets quantized to whole cache lines in each cache.  All cache lines
   455         *  within range are invalidated in L1P cache.  All cache lines within the
   456         *  range are written back to L2 or external and then invalidated in L1D
   457         *  cache. All cache lines within the range are written back to external
   458         *  memory and then invalidated in L2 cache.
   459         */
   460        override Void wbInv(Ptr blockPtr, SizeT byteCnt, Bits16 type, Bool wait);
   461    
   462        /*!
   463         *  ======== invL1pAll ========
   464         *  Invalidate all of L1 Program cache
   465         *
   466         *  Performs a global invalidate of L1P cache.
   467         *  Polls the L1P invalidate register until done.
   468         */
   469        Void invL1pAll();
   470    
   471        /*!
   472         *  ======== wbAll ========
   473         *  Write back all caches
   474         *
   475         *  Perform a global write back.  There is no effect on L1P cache.
   476         *  All cache lines are left valid in L1D cache and dirty lines in L1D cache
   477         *  are written back to L2 or external.  All cache lines are left valid in
   478         *  L2 cache and dirty lines in L2 cache are written back to external.
   479         *  This function does not wait for write back operation to perculate
   480         *  through the whole memory system before returing. Call Cache_wait(),
   481         *  after this function if necessary.
   482         */
   483        override Void wbAll();
   484    
   485        /*!
   486         *  ======== wbL1dAll ========
   487         *  Write back L1D cache
   488         *
   489         *  Perform a global write back of L1D cache. There is no effect on L1P
   490         *  or L2 cache.  All cache lines are left valid in L1D cache and the
   491         *  dirty lines in L1D cache are written back to L2 or external.
   492         *  This function does not wait for write back operation to perculate
   493         *  through the whole memory system before returing. Call Cache_wait(),
   494         *  after this function if necessary.
   495         */
   496        Void wbL1dAll();
   497    
   498        /*!
   499         *  ======== wbInvAll ========
   500         *  Write back invalidate all caches
   501         *
   502         *  Performs a global write back and invalidate.  All cache lines are
   503         *  invalidated in L1P cache.  All dirty cache lines are written back to L2
   504         *  or external and then invalidated in L1D cache.  All dirty cache lines
   505         *  are written back to external and then invalidated in L2 cache.
   506         *  This function does not wait for write back operation to perculate
   507         *  through the whole memory system before returing. Call Cache_wait(),
   508         *  after this function if necessary.
   509         */
   510        override Void wbInvAll();
   511    
   512        /*!
   513         *  ======== wbInvL1dAll ========
   514         *  Write back invalidate L1D cache
   515         *
   516         *  Performs a global write back and invalidate of L1D cache.
   517         *  All dirty cache lines are written back to L2 or
   518         *  external and then invalidated in L1D cache.
   519         *  This function does not wait for write back operation to perculate
   520         *  through the whole memory system before returing. Call Cache_wait(),
   521         *  after this function if necessary.
   522         */
   523        Void wbInvL1dAll();
